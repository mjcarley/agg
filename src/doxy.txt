/**
 * @mainpage AGG: a library for Aerodynamic Geometry Generation
 *
 * AGG is a library for generating "aerodynamic" geometries, based on
 * the modified and extended version of the CST method of Brenda Kulfan.
 *
 */

/**
 * @defgroup bernstein Bernstein polynomials
 *
 * @brief Evaluation of Bernstein polynomials
 *
 */

/**
 * @defgroup sections Section geometries
 *
 * @brief Evaluation of (cross-) sections such as ellipses and aerofoils
 *
 */

/**
 * @defgroup transforms Transformations
 *
 * @brief Geometrical transformation of cross-section shapes
 *
 */

/**
 * @defgroup surfaces Surfaces
 *
 * @brief Surfaces composed of transformed sections
 *
 */

/**
 * @defgroup variables Variables and expressions
 *
 * @brief Symbolically defined variables and expressions
 *
 */

/**
 * @defgroup shapes Section shapes
 *
 * @brief Built-in and user-defined section shapes
 *
 */

/**
 * @defgroup patches Surface patches
 *
 * @brief Patches used to map parts of surfaces and generate meshes
 *
 */

/**
 * @defgroup intersections Surface intersections
 *
 * @brief Evaluation of surface intersections
 *
 */


/**
 * @defgroup mesh Surface meshes
 * 
 * @brief Generation of watertight meshes of collections of surfaces
 *
 */

/**
 * @defgroup body General bodies
 * 
 * @brief Bodies are made up of collections of elementary surfaces
 *
 */

/**
 * @defgroup blends Surface blends
 * 
 * @brief Intersecting bodies are joined using smooth surface blends
 *
 */
